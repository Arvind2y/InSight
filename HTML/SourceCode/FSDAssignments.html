    var hour = currentDateTime.getHours();
    var minute = currentDateTime.getMinutes();
    var second = currentDateTime.getSeconds();
    var temp = ((hour > 12) ? hour - 12 : hour);

    temp = '12';

Solution:
function getCharacterCount(inputstring) {

    var charArray = inputstring.replace(/ /g, '').split("");
    var countedChars = charArray.reduce(function (charsArr, curChar) {
        if (curChar in charsArr) {
            charsArr[curChar]++;
        }
        else {
            charsArr[curChar] = 1;
        }
        return charsArr;
    }, {});

    var result = Object.keys(countedChars).map(function (key) {
        return ' "' + key + '":' + countedChars[key];
    });

    console.log(result);
    document.querySelector("#lblCharCount").innerHTML = ` ${result}`;
}

function randomString(ranLength) {
    var chars = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789";

17. Write a JavaScript function to find longest substring in a given a string without repeating characters. Sample Input: google.com Sample Output:gle.com Sample input : example.com Sample output:xample.co

Solution:
function longestWord(input) {
    var chars = input.split('');
    var curr_char;
    var str = "";
    var longest_string = "";
    var hash = {};
    for (var i = 0; i < chars.length; i++) {
        curr_char = chars[i];
        if (!hash[chars[i]]) {
            str += curr_char;
            hash[chars[i]] = { index: i };
        }
        else {
            if (longest_string.length <= str.length) {
                longest_string = str;
            }
            var prev_dupeIndex = hash[curr_char].index;
            var str_FromPrevDupe = input.substring(prev_dupeIndex + 1, i);
            str = str_FromPrevDupe + curr_char;
            hash = {};
            for (var j = prev_dupeIndex + 1; j <= i; j++) {
                hash[input.charAt(j)] = { index: j };
            }
        }
    }
    var restult = longest_string.length > str.length ? longest_string : str;
    document.querySelector("#lblLongestStr").innerHTML = ` ${restult}`;
}

document.querySelector("#btnLongestStr").addEventListener("click", () =>
    longestWord(document.querySelector("#txtLongestStr").value), true);
