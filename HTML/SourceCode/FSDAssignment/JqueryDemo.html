Solution:
$(document).ready(function () {
    $("a[href='#top']").click(function () {
        $("html, body").animate({ scrollTop: 0 }, {
            opacity: "show"
          }, {
            duration: "slow",
            easing: "easein"
          });
        return false;
    });

    
});

4. Add 3 new options to a drop-down list using jQuery. 

                                        




Add options to a drop-down list using jQuery.



List of Colors :

Solution:
document.querySelector("#btnAddOptions").addEventListener("click", () =>
addColorOption(), true);

function addColorOption() {
    var myOptions = {
        val1: 'Blue',
        val2: 'Orange',
        val3: 'Aqua'
    };
    var mySelect = $('#myColors');
    $.each(myOptions, function (val, text) {
        mySelect.append(
            $('').val(val).html(text)
        );
    });
}
